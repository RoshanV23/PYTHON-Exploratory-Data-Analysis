Multiplying the Size value by 1,000, if size is mentioned in Mb¶

In [12]:
def change(Size):
    if 'M' in Size:
        x=Size[:-1]
        x=float(x)*1000
        return x
    elif'k'in Size:
        x=x=Size[:-1]
        x=float(x)*1000
        return x
    else:
        return None
